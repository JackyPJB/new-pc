1.pip升级 
python -m pip install Cupgrade pip
2.换源 
在windows资源管理器下键入%APPDATA% 
新建文件夹pip 
进入新建的这个pip文件夹内，新建文件pip.ini，以记事本形式打开，然后敲入 
[global]  
timeout = 6000 
index-url = http://pypi.douban.com/simple  
trusted-host = pypi.douban.com
